    () => {
        unsafe {
            asm!("int3"::);
            // For whatever reason, debuggers will show the paused line as the
            // next one after the int3, so we have this dummy line here for
            // ergonomics.
            let _ = 1;
        }
    };
    ($e:expr) => {
        if $e {
            breakpoint!();
        }
    };
